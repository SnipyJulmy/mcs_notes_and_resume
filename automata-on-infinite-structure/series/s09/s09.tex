\section*{Exercise 1}

\subsection*{1 : $(ab^*)^\omega$}

\[
  0 \in Q_a \wedge \forall x.(x \in Q_a \to \exists y.(y \in Q_a \wedge x < y))
\]

\subsection*{2 : $(aaa)^+(a+b)^\omega$}

\[
  0 \in Q_a \wedge
  0 + 1 \in Q_a \wedge
  0 + 1 + 1 \in Q_a \wedge
  \forall x.( 0 + 1 + 1 < x \to (x \in Q_a \vee x \in Q_b))
\]
